87  uint64_t OffTimer :1;

95  uint64_t :8;
96 
97  // Byte 24
98  uint64_t SwingV :4; // 0000 = off, 1111 = on
99  uint64_t Fan :4;

102  uint64_t :4;

105  uint64_t OffTime :12; // timer mins past midnight
106  // Byte 29
107  uint64_t Powerful :1;
