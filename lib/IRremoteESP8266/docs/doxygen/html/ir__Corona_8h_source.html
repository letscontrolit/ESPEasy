43 union CoronaProtocol {
44  uint8_t raw[kCoronaAcStateLength];
45  CoronaSection sections[kCoronaAcSections];
46  struct {
47  // Byte 0
48  uint8_t :8;
49  // Byte 1
50  uint8_t :8;
51  // Byte 2
52  uint8_t :8;
53  // Byte 3
54  uint8_t Fan :2;
55  uint8_t :1;
56  uint8_t Econo :1;
57  uint8_t :1; // always on
58  uint8_t :1;
59  uint8_t SwingVToggle :1;
60  uint8_t :1;
61  // Byte 4
62  uint8_t :8;
63  // Byte 5
64  uint8_t Temp :4;
65  uint8_t Power :1;
66  uint8_t PowerButton :1;
67  uint8_t Mode :2;
68  };
