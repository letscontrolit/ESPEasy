90 union CoolixProtocol {
91  uint32_t raw;
92  struct { // Only 24 bits are used.
93  // Byte
94  uint32_t :1; // Unknown
95  uint32_t ZoneFollow1:1;
96  uint32_t Mode :2;
97  uint32_t Temp :4;
98  // Byte
99  uint32_t SensorTemp :5;
100  uint32_t Fan :3;
101  // Byte
102  uint32_t :3; // Unknown
103  uint32_t ZoneFollow2:1;
104  uint32_t :4;
105  };
106 };
